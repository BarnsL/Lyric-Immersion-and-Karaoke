# PyInstaller version resource (VSVersionInfo). Embeds real Windows file metadata
# (company / product / description / version / copyright) into the .exe. An exe
# with NO version resource is a strong Windows Defender / SmartScreen heuristic
# signal ("unsigned + no metadata + packed" → flagged on clean machines). This
# does NOT replace code-signing, but it measurably lowers the false-positive rate
# and gives the app a real identity in the UAC / SmartScreen dialogs.
#
# Keep the four version numbers in sync with version.py (__version__) on release:
# filevers/prodvers = (MAJOR, MINOR, PATCH, 0).
VSVersionInfo(
  ffi=FixedFileInfo(
    filevers=(1, 1, 54, 0),
    prodvers=(1, 1, 54, 0),
    mask=0x3f,
    flags=0x0,
    OS=0x40004,
    fileType=0x1,
    subtype=0x0,
    date=(0, 0),
  ),
  kids=[
    StringFileInfo(
      [
        StringTable(
          "040904B0",
          [
            StringStruct("CompanyName", "Purple Industries"),
            StringStruct("FileDescription", "Lyric Immersion and Karaoke — synced furigana/romaji/English lyric overlay"),
            StringStruct("FileVersion", "1.1.54.0"),
            StringStruct("InternalName", "Lyric-Immersion-and-Karaoke"),
            StringStruct("LegalCopyright", "Copyright (c) Purple Industries"),
            StringStruct("OriginalFilename", "Lyric-Immersion-and-Karaoke.exe"),
            StringStruct("ProductName", "Lyric Immersion and Karaoke"),
            StringStruct("ProductVersion", "1.1.54.0"),
          ],
        )
      ]
    ),
    VarFileInfo([VarStruct("Translation", [1033, 1200])]),
  ],
)
